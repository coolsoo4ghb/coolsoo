Zemax macro example


# distortion model
Print "distortion model generation"

Hx1 = -0.629
Hx2 = 0.629
delhx = Hx2-Hx1
Nstepx = 10

Hy1 = -0.778
Hy2 = -0.5
delhy = Hy2-Hy1
Nstepy = 5

format .10
For i,0,Nstepx,1
For j,0,Nstepy,1
  Hx = Hx1 + delhx/Nstepx*i
  Hy = Hy1 + delhy/Nstepy*j

  # raytrace for chief ray
  RAYTRACE Hx,Hy,0,0
  x = RAYX(21)
  y = RAYY(21)
  lx = RAYL(0)
  ly = RAYM(0)
  lz = RAYN(0)
  PRINT "x= ",x,"y= ",y,"lx= ",lx,"ly= ",ly,"lz= ",lz

Next
Next
