aaa = zeros(235,3); % x-data, y-data, value-data

% cc cv data to aaa

% interpolation function generation
F = scatteredInterpolant(aaa(:,1),aaa(:,2),aaa(:,3));
F.Method = 'linear';
F.ExtrapolationMethod = 'none';

maxX = max(aaa(:,1)); minX = min(aaa(:,1));
nX = 100; delX = (maxX-minX)/(nX-1);
maxY = max(aaa(:,2)); minY = min(aaa(:,2));
nY = 200; delY = (maxY-minY)/(nY-1);

x = minX:delX:maxX;
%y = minY:delY:maxY;
y = maxY:-delY:minY; % for simple interpretation of imagesc(zz) 
                     %  in x-axis : column, y-axis : line match

[xx,yy] = meshgrid(x,y);

% interpolation process
zz = zeros(size(xx));
for ic=1:length(x); % column (x)
    for il=1:length(y); % line(y)
        zz(il,ic) = F(x(ic),y(il));
    end;
end;

%figure; mesh(xx,yy,zz);
%figure; surf(xx,yy,zz);

figure; contourf(xx,yy,zz,'LineStyle','none');
xlabel('x'); ylabel('y');colorbar;

figure; imagesc(zz); % to be cautious for the match of x, y axis
xlabel('--> x'); ylabel('--> y');colorbar;

figure; imagesc(x,y,zz);
xlabel('x'); ylabel('y');colorbar;

figure; imagesc(x,y,zz);
xlabel('x'); ylabel('y');colorbar;
set(gca,'YDir','normal')
