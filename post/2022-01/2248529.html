matlab fit(), fittype()

% curve fitting : y = A*(1 - exp(-b*t))
%  find A & b
%
clear;
t = [01.046.068.06]';
y = [0-1.099408-2.02742-2.214568]';

ft = fittype('A*(1-exp(-b*t))','independent',{'t'}, ...
     'coefficients',{'A','b'});
options=fitoptions('Method','NonlinearLeastSquares');
options.StartPoint=[-1.0  1.0];
[c2, gof2,oput2] = fit(t,y,ft,options);
%[c2, gof2,oput2] = fit(t,y,ft);
c2 % print curve fitted paramters

% plot
figure;
plot(t,y,'ob')
tt=[0:0.1:10]';
yy=ft(c2.A,c2.b,tt);
hold on
plot(tt,yy,'r')
